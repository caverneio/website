```ts showLineNumbers title="src/pages/api/posts/[id]/images.ts"  {6-10, 13}
let images = [];

let i = 0;
while (true) {
  let file = form_data.get(`file-${i}`);
  let alt = form_data.get(`alt-${i}`);

  if (!file) {
    break;
  }

  file = file as File;

  let fileName: string = file.name;
  let mimeType = file.type;
  let fileData = Buffer.from(await file.arrayBuffer());

  images.push({
    alt: alt ? alt.toString() : null,
    file: {
      name: fileName,
      mediaType: mimeType,
      base64Content: fileData.toString("base64"),
    },
    status: newStatus.id,
  });

  i++;
}

await xata.db.image.create(images);
```
